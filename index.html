Vincular Planilha Real do Google Sheets

Com o GestaSheet, você pode ligar sua planilha real do Google diretamente a esta tela! Ao salvar novas entradas ou deletar clientes, os dados são sincronizados utilizando a API integrada por Google Apps Script.

Como configurar sua planilha em 2 minutos (Passo a Passo)

  1. Abra a sua planilha no Google Sheets (Aba ativa com nome CLIENTES).
  2. No menu superior, clique em Extensões > Apps Script.
  3. Apague todo o código padrão e cole o script abaixo.
  4. Clique em Implantar > Nova implantação.
  5. Selecione App da Web em Configurações.
  6. Em Quem tem acesso, marque Qualquer pessoa.
  7. Conclua a implantação, autorize as permissões e cole o link acima!
function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("CLIENTES");
  if(!sheet) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: "Crie uma aba com o nome 'CLIENTES'"})).setMimeType(ContentService.MimeType.JSON);
  }
  var data = sheet.getDataRange().getValues();
  var clients = [];
  
  for(var i = 1; i < data.length; i++) {
    if(!data[i][0]) continue;
    clients.push({
      id: i,
      cliente: data[i][0],
      mensalidade: parseFloat(data[i][1]) || 0,
      observacao: data[i][2] || "",
      contato: data[i][3] || "",
      cidade: data[i][4] || "",
      situacao: data[i][5] || "ATIVO",
      regime: data[i][6] || "",
      emite: data[i][7] || "",
      sistema: data[i][8] || "",
      versao: data[i][9] || "",
      cnpj: data[i][10] || "",
      contabilidade: data[i][11] || "Não Atribuída",
      pasta: data[i][12] || ""
    });
  }
  return ContentService.createTextOutput(JSON.stringify(clients))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("CLIENTES");
    
    if(payload.action === "syncAll") {
      sheet.clearContents();
      var headers = ["CLIENTES", "MENSALIDADE", "OBSERVAÇÃO", "CONTATO", "CIDADE", "SITUAÇÃO", "REGIME", "EMITE?", "SISTEMA", "VERSÃO", "CNPJ", "CONTABILIDADE", "PASTA"];
      sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
      
      var rows = payload.clients.map(function(c) {
        return [
          c.cliente,
          c.mensalidade,
          c.observacao || "",
          c.contato || "",
          c.cidade || "",
          c.situacao || "ATIVO",
          c.regime || "",
          c.emite || "",
          c.sistema || "",
          c.versao || "",
          c.cnpj || "",
          c.contabilidade || "Não Atribuída",
          c.pasta || ""
        ];
      });
      
      if(rows.length > 0) {
        sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
      }
      return ContentService.createTextOutput(JSON.stringify({status: "success"}))
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
